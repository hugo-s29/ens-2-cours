  \chapter{Protocoles interactifs.}

  Deux machines discutent : le \textit{prouveur} $\prov$ et le \textit{vérificateur} $\verif$.
  Le prouveur a une puissance de calcul illimitée, et le vérificateur ne fait pas initialement "confiance" au prouveur.

  \begin{exm}
    On donne un protocole interactif pour \pbSat.

    \begin{algorithm}[H]
      \caption{Protocole interactif pour \pbSat}
      \begin{paralgo}{2}[Vérificateur $\verif$, Prouveur $\prov$]
        \State[1]{Calculer $(a_1, \ldots, a_n)$ une assignation qui satisfait $F$.}
        \Message{1}{0}{$a_1,\ldots,a_n$}
        \If[0]{$F(a_1, \ldots, a_n) = \texttt{1}$}
        \State[0]{\textbf{Accepter}}
        \Else[0]
        \State[0]{\textbf{Rejeter}}
        \EndIf[0]
      \end{paralgo}
    \end{algorithm}

    Analysons ce protocole :
    \begin{itemize}
      \item si $F \not\in \pbSat$ alors $\verif$ rejette toujours, peu importe $\tilde{\prov}$ ;
      \item si $F \in \pbSat$ alors $\prov$ peut convaincre $\verif$ d'accepter.
    \end{itemize}
  \end{exm}

  \begin{exm}
    On donne un protocole interactif pour le \textit{non-isomorphisme de graphes}.

    On définit le problème \pbIsoGraph :
    \showIsoGraph

    où un \textit{isomorphisme} entre $(W, E)$ et $(W, E') = \pi(W, E)$ est une permutation $\pi : W \to W$ des sommets avec
    \[
      E' := \mleft\{\,(i,j) \;\middle|\; \big(\pi(i), \pi(j)\big) \in E\,\mright\} 
    .\]
    On notera alors $(W, E) \cong (W, E')$.

    On a que $\pbIsoGraph \in \np$ (avec $\pi$ le certificat).
    Mais, on ne sait pas si le problème est $\np$-complet, il existe un algorithme en temps $n^{\mathrm{O}(\log n)}$.
    (Par définition, le problème de \textit{non}-isomorphisme de graphe est dans $\conp$.)

    On donne un protocole interactif pour le problème de non-isomorphisme de graphe.
    En entrée, $G_0, G_1$ est une paire de graphes sur un même ensemble de sommets $W$.

    Le but de $\prov$ est de convaincre $\verif$ que $G_0 \ncong G_1$.
    \begin{itemize}
      \item Si $G_0 \ncong G_1$ alors $G'$ est isomorphe à un unique graphe $G_{b'}$ et renvoie $b'$ au vérificateur.
      \item Si $G_0 \cong G_1$, alors pour tout prouveur, $\Pr[\text{$\verif$ accepte}] = 1/2$.
        En effet, pour tout $G'$, 
        \[
          \Pr[\text{$\tilde{\prov}$ renvoie $G'$}  \mid b = 0] = \Pr[\text{$\tilde{\prov}$ renvoie $G'$}  \mid b = 1]
        ,\]
        notons $p_{i,j} := \Pr[\text{$\tilde{\prov}$ envoie $b' = j$}  \mid  b = i]$ et on observe que $p_{11} = p_{21}$, d'où 
        \[
          \Pr[\text{$\verif$ accepte}] = \frac{1}{2} (p_{11} + p_{22}) = \frac{1}{2}(p_{21} + p_{22}) = \frac{1}{2}
        .\]
    \end{itemize}

    \begin{algorithm}[H]
      \caption{Protocole interactif pour $\Sigma^\star \setminus \pbIsoGraph$}
      \begin{paralgo}{2}[Vérificateur $\verif$, Prouveur $\prov$]
        \Require{$G_0, G_1$}
        \State[0]{Choisir un bit $b \in \{0,1\}$ aléatoire}
        \State[0]{Choisir une permutation $\pi$ de $W$ aléatoire}
        \State[0]{Calculer $G' := \pi(G_b)$}
        \Message{0}{1}{$G'$}
        \If[1]{$G_0 \ncong G_1$}
        \State[1]{Le graphe $G'$ est isomorphe à un unique $G_{b'}.$}
        \Else[1]
        \State[1]{Soit $b' \in \{0,1\} $ choisi aléatoirement.}
        \EndIf[1]
        \Message{1}{0}{$b'$}
        \If[0]{$b = b'$}
        \State[0]{\textbf{Accepter}}
        \Else[0]
        \State[0]{\textbf{Rejeter}}
        \EndIf[0]
      \end{paralgo}
    \end{algorithm}

    On peut réduire l'erreur en répétant le protocole $k$ fois, et on accepte si $(G_0, G_1)$ est toujours accepté par $\verif$, et on aura $\Pr[\text{$\verif$ accepte}] \le 1 / 2^k$.
  \end{exm}

  \section{La classe $\ip$.}

  On a la situation représentée en figure~\ref{fig:protocole-interactif}, où
  \begin{itemize}
    \item le $x$ est l'entrée du problème (de taille $n$) ;
    \item le $r$ est la donnée des choix aléatoires de $v$ (de taille polynomiale en $n$) ;
    \item les $y_i$ sont les messages de $\verif$ à $\prov$ (de taille polynomiale en $n$) ;
    \item les $z_i$ sont les messages de $\prov$ à $\verif$ (de taille polynomiale en $n$) ;
    \item le $s$ est la longueur de l'échange (avec $s$ polynomial en $n$).
  \end{itemize}
  La puissance de calcul de $\prov$ est \textit{non-bornée}, et celle de $\verif$ est polynomialement bornée.
  La \textit{décision finale du vérificateur} est $\verif(x, r, z_1, \ldots, z_s) \in \{\texttt{0},\texttt{1}\}$.

  \begin{defn}
    On définit $\intro*\ip$ la classe des langages que l'on peut reconnaitre à
    l'aide d'un protocole interactif avec probabilité d'erreur inférieur à
    $1/3$. Autrement dit, un langage $L$ est dans $\ip$ ss'il existe un
    vérificateur $\intro*\verif$ tel que
     \begin{enumerate}
      \item si $x \in L$ alors il existe un prouveur $\intro*\prov$ tel que
        \[
          \Pr[\text{$\prov$ pousse $\verif$ à accepter}] \ge  \frac{2}{3} \;
        ;\]
      \item si $x \not\in L$ alors pour tout prouveur $\tilde{\prov}$, 
        \[
          \Pr[\text{$\tilde{\prov}$ pousse $\verif$ à accepter}] \le \frac{1}{3} \;
        .\]
    \end{enumerate}
  \end{defn}

  \begin{exm}
    On a $\pbSat \in \ip$ et $\Sigma^\star \setminus \pbIsoGraph \in \ip$.
  \end{exm}

  \begin{thm}
    On a $\ip = \pspace$.
  \end{thm}

  \begin{rmk}
    On a $\ip \subseteq \pspace$, c'est le "sens facile".

    En revanche, le résultat $\pspace \subseteq \ip$ \textit{\textbf{ne se
    relativise pas}} : il existe $A$ tel que $\oracle{\pspace} A$ n'est pas
    inclus dans $\oracle{\ip} A$ (et même~$\oracle{\conp} A$ n'est pas inclus
    dans $\oracle{\ip} A$).
  \end{rmk}

  Pour montrer $\pspace \subseteq \ip$, on utilise le fait que \pbQBF\ est
  $\pspace$-complet : il suffira donc de montrer que $\pbQBF \in \ip$, et on
  pourra utiliser les propriétés spécifiques de \pbQBF, en particulier
  l'\textit{\textbf{arithmétisation}} (on peut voir des formules booléennes
  comme des polynômes).

  \begin{defn}
    On définit le problème \pbSharpSatD\ par 

    \showSharpSatD

    ou, de manière équivalente,
    \[
      \pbSharpSatD := \mleft\{\,\code{\phi, k} \;\middle|\; 
        \begin{array}{l}
          \text{$\phi$ est une formule booléenne satisfaite}\\
          \text{par exactement $k$ assignations}
        \end{array}
      \,\mright\} 
    .\] 
  \end{defn}

  \begin{thm}
    On a $\pbSharpSatD \in \ip$.
  \end{thm}
  \begin{prv}
    La preuve se base sur la technique d'\textit{\textbf{arithmétisation}} : on
    associe à $\phi(x_1, \ldots, x_n)$ un polynôme $P_\phi(x_1, \ldots, x_n)$
    que l'on définit par induction :
    \begin{itemize}
      \item si $\phi = x$, alors $P_\phi := x$ ;
      \item si $\phi = \psi \wedge \vartheta$, alors $P_\phi := P_\psi \times P_\vartheta$ ;
      \item si $\phi = \lnot \psi$ alors  $P_\phi := 1 - P_\psi$ ;
      \item si $\phi = \psi \vee \vartheta$, alors
        \[
          P_\phi := P_\psi + P_\vartheta - P_\psi \times P_\vartheta = 1 - (1 - P_\psi) \times (1 - P_\vartheta)
        .\]
    \end{itemize}
    On définit également, pour $i \in \llbracket 0, m\rrbracket$,
    \[
    f_i(x_1, \ldots, x_i) := \sum_{(a_{i+1}, \ldots, a_n) \in \{0, 1\}^{n - i}} P_\phi(x_1, \ldots, x_i, a_{i+1}, \ldots, a_n)
    .\]
    En particulier,  $f_n(x_1, \ldots, x_n) = P_\phi(x_1, \ldots, x_n)$.
    Et, de plus, $f_0$ est une constante qui compte le nombre d'instances qui
    vont satisfaire la formule $\phi$. On a \[
    \deg f_i \le \deg P_\phi \le |\phi|
    ,\] et si $f \neq g$ avec $\deg f \le d$ et $\deg g \le d$ alors
    \begin{equation}
      \label{prv:sharpsat-in-ip-neq-poly}
      \tag{$*$}
      \Pr[f(x) = g(x)] \le \frac{d}{|S|}
    ,\end{equation}
    si $x$ est choisi uniformément dans $S$.

    Maintenant, décrivons le protocole.

    \begin{algorithm}[H]
      \caption{Protocole interactif pour \pbSharpSatD}
      \begin{paralgo}{2}[Vérificateur $\verif$, Prouveur $\prov$]
        \State[1]{Calculer des polynômes $f_0, \ldots, f_n$}
        \Message{1}{0}{$f_0$}
        \If[0]{$f_0 \neq k$}
        \State[0]{\textbf{Rejeter}}
        \EndIf[0]
        \Message{1}{0}{$f_1(z)$}
        \State[0]{$y_0 := f_1(0) + f_1(1)$}
        \If[0]{$f_0 \neq y_0$}
        \State[0]{\textbf{Rejeter}}
        \EndIf[0]
        \State[0]{Choisir $r_1 \in S$ aléatoirement}
        \Message{0}{1}{$r_1$}
        \State[1]{Calculer $f_2(r_1, z)$, polynôme en $z$}
        \Message{1}{0}{$f_2(r_1, z)$}
        \State[0]{$y_1 := f_2(r_1, 0) + f_2(r_1, 1)$}
        \If[0]{{$f_1(r_1) \neq y_1$}}
        \State[0]{\textbf{Rejeter}}
        \EndIf[0]
        \State[0]{Choisir $r_2 \in S$ aléatoirement}
        \Message{0}{1}{$r_2$}
        \State[1]{Calculer $f_3(r_1, r_2, z)$, polynôme en $z$}
        \Message{1}{0}{$f_3(r_1, r_2, z)$}
        \State{\textit{etc}}
        \State[0]{$y_n := f_n(r_1, \ldots, r_n)$}
        \State[0]{\textbf{Accepter} si \rlap{$P_\phi(r_1,\ldots, r_n) = y_n$}}
        \State[0]{\textbf{Rejeter} sinon}
      \end{paralgo}
    \end{algorithm}

    Dans un premier temps, analysons la complétude du protocole. Si
    $\code{\phi, k} \in \pbSharpSatD$, montrons qu'il existe $\prov$ qui va
    convaincre  $\verif$ à coup sûr. Pour cela, il suffit de prendre $\prov$
    qui respecte le protocole avec les $f_i$ définies précédemment.

    Maintenant, regardons la consistance du protocole. Supposons que
    $\code{\phi, k} \not\in \pbSharpSatD$, \textit{i.e.}, $k$ n'est pas le
    nombre d'assignations qui vont satisfaire $\phi$. Montrons que tout
    prouveur $\tilde{\prov}$ a probabilité inférieure à $1/3$ de convaincre
    $\verif$. Lors du protocole,  $\tilde{\prov}$ va envoyer des polynômes
    $\tilde{f}_i$ qui peuvent être différents des $f_i$. On dira que
    $\tilde{\prov}$ est \textit{chanceux à la phase $i$} si
    \[
    f_i(r_1, \ldots, r_{i-1}, z) \neq  \tilde f_i (r_1, \ldots, r_{i-1}, z)
    \] mais que $\verif$ choisit $r_i$ tel que \[
    f_i(r_1, \ldots, r_i) = \tilde f_i (r_1, \ldots, r_i)
    .\] 

    \begin{lem}
      Si $\code{\phi, k} \not\in \pbSharpSatD$ et si $\tilde{\prov}$ n'est jamais chanceux alors $\verif$ rejette.
    \end{lem}
    \begin{prv}
      On a que $k = \tilde f_0 \neq f_0$ sinon $\verif$ rejette immédiatement.
      De même, $\tilde f_0 = \tilde f_1 (0) + \tilde f_1 (1)$ sinon $\verif$
      rejette à la phase suivante.
      On a nécessairement que $\tilde f_1 \neq f_1$ car, sinon, on aurait
      \[
      \tilde f_0 = \tilde f_1 (0) + \tilde f_1 (1) = f_1(0) + f_1(1) = f_0
      .\]
      Puisque $\tilde{\prov}$ n'est pas chanceux, $\verif$ va choisir une
      valeur de $r_1$ telle que $\tilde f_1 (r_1) \neq f_1(r_1)$. On peut
      raisonner ainsi jusqu'à la fin du protocole. Lors de la dernière phase, on a donc \[
      \tilde f_n(r_1, \ldots, r_n) \neq f_n(r_1, \ldots, r_n)
      ,\]
      et, en particulier, $\tilde f_n(r_1, \ldots, r_n) \neq P_\phi(r_1,
      \ldots, r_n)$. Ainsi, $\verif$ va finalement rejeter (s'il n'avait pas
      rejeté avant).
    \end{prv}

    \begin{lem}
      On a
      \[
        \Pr[\exists i, \text{$\tilde {\prov}$ est chanceux à l'étape $i$}] \le \frac{n \times |\phi|}{|S|}
      .\]
    \end{lem}
    \begin{prv}
      Pour $i \in \llbracket 1, n\rrbracket$, on a par (\ref{prv:sharpsat-in-ip-neq-poly}) que
      \[
        \Pr[\text{$\tilde {\prov}$ est chanceux à l'étape $i$}] \le \frac{|\phi|}{|S|}
      ,\]
      et il suffit juste d'appliquer la borne de l'union.
    \end{prv}

    Désormais, il ne reste plus qu'à analyser la complexité pour vérifier que
    le vérificateur est bien en temps polynomial. Par la suite, on supposera
    que les polynômes sont à coefficients entiers. On peut vérifier que les
    polynômes échangés ont des coefficients de taille polynomiale. Sinon, on
    peut se placer sur un corps fini $\mathds{F}$ dans lequel les coefficients
    sont nécessairement de taille bornée. La seule contrainte sur $\mathds{F}$
    est $|\mathds{F}| \ge |S|$.
    
    Le prouveur $\prov$ peut envoyer $p$ tel que $\mathds{F} = \mathds{Z} / p
    \mathds{Z}$ à $\verif$ qui va vérifier que $p$ est premier, avant toute
    chose, ce qui se fait en temps polynomial.\footnote{Notamment avec le test
    Agrawal–Kayal–Saxena.} Pour tout $S$, on a qu'il existe un nombre premier
    $p$ avec $|S| \le p \le 2 |S|$ qui peut être renvoyé par $\prov$
    (\textit{postulat de Bertrand}).

    Il ne reste plus qu'à représenter $P_\phi$.
    Puisque le représenter avec des monômes peut être exponentiel (par exemple
    avec la formule~$\phi = \bigwedge_{i=1}^n (\lnot x_i)$), on peut le
    représenter avec un circuit arithmétique ou par une formule arithmétique
    (\textit{i.e.}\ un arbre).
  \end{prv}

  \section{Preuve de $\ip \subseteq \pspace$.}

  Soit $L \in \ip$ et soit $\verif$ le vérificateur associé.

  \begin{lem}
    On peut calculer, sur une entrée $x$, en espace polynomial
    \[
      \max_{\prov} \Pr[\text{\prov\ pousse \verif\ à accepter $x$}]
    .\]
  \end{lem}
  \begin{prv}
    On suppose que sur une entrée $x \in \{\texttt{0}, \texttt{1}\}^n$,
    \begin{itemize}
      \item le nombre de "phases" est exactement $s := s(n)$ ; 
      \item le vérificateur $\verif$ utilise exactement $r := r(n)$ bits aléatoires à chaque phase ;
      \item le prouveur $\prov$ envoie des messages de longueurs exactement $\ell := \ell(n)$ à chaque phase.
    \end{itemize}
    On peut représenter cette situation comme un arbre.

    \begin{figure}[H]
      \centering
      \includegraphics[width=0.85\textwidth,clip, trim=2cm 16cm 2.5cm 1cm]{figures/probability_accepting.pdf}
      \caption{Arbre permettant de calculer le résultat}
    \end{figure}

    En chaque nœud on a une histoire partielle $H$ qui correspond à la liste des messages envoyés par $\prov$ et des bits aléatoires tirés par $\verif$.
    La \textit{valeur} d'un nœud est 
    \[
      \max_{\prov} \Pr[\text{\prov\ pousse \verif\ à accepter $x$}]
    .\]
    Pour une feuille, sa valeur est $0$ ou $1$. 
    \begin{enumerate}
      \item Pour un nœud $\prov$, on a 
         \[
           \Pr[\text{\verif\ accepte $x$}  \mid  H] = \max_{m \in \{\texttt{0},\texttt{1}\}^n } \Pr[\text{\verif\ accepte $x$}  \mid  H \texttt{\#} m]
        .\] 
      \item Pour un nœud $\verif$, on a
         \[
           \Pr[\text{\verif\ accepte $x$}  \mid  H] = \frac{1}{2^r} \sum_{y \in \{\texttt{0},\texttt{1}\}^r} \Pr[\text{\verif\ accepte $x$}  \mid  H \texttt{\#} y]
        .\]
    \end{enumerate}
    On évalue la racine avec un algorithme récursif, avec une alternance de $\max$ et de moyenne.
  \end{prv}

  \begin{crlr}
    On peut toujours supposer que $\prov$ fonctionne avec un espace polynomial.
    \qed
  \end{crlr}
